async function loadLeaderboard() { const subject = document.getElementById('subject-filter').value; if (!subject) { document.getElementById('leaderboard-placeholder').classList.remove('hidden'); document.getElementById('leaderboard-table').classList.add('hidden'); return; } const mode = 'challenge'; const key = getResultsKey(subject, mode); const tbody = document.getElementById('leaderboard-body'); const table = document.getElementById('leaderboard-table'); const placeholder = document.getElementById('leaderboard-placeholder'); const emptyMsg = document.getElementById('leaderboard-empty'); const info = document.getElementById('leaderboard-info'); tbody.innerHTML = 'Đang tải...'; table.classList.remove('hidden'); placeholder.classList.add('hidden'); emptyMsg.classList.add('hidden'); info.classList.add('hidden'); if (!KV_REST_API_URL || !KV_REST_API_TOKEN) { tbody.innerHTML = 'Chưa cấu hình Redis'; return; } try { const res = await fetch(`${KV_REST_API_URL}/get/${key}`, { headers: { Authorization: `Bearer ${KV_REST_API_TOKEN}` } }); let results = []; if (res.ok) { const data = await res.json(); if (data.result === null || data.result === undefined) { results = []; } else { try { const parsed = JSON.parse(data.result); results = Array.isArray(parsed) ? parsed : []; } catch (parseErr) { console.error("Lỗi parse JSON từ Upstash:", parseErr, data.result); results = []; } } } else if (res.status === 404) { results = []; } else { throw new Error(`HTTP ${res.status}: ${await res.text()}`); } if (results.length === 0) { emptyMsg.classList.remove('hidden'); table.classList.add('hidden'); info.classList.add('hidden'); return; } // Bây giờ results chắc chắn là array → sort an toàn results.sort((a, b) => b.score - a.score || a.timeSeconds - b.timeSeconds); tbody.innerHTML = ''; let rank = 1; results.slice(0, 50).forEach(r => { const tr = document.createElement('tr'); tr.className = rank <= 3 ? 'bg-yellow-50' : ''; tr.innerHTML = ` ${rank} ${r.name}${r.name === studentName ? ' (Bạn)' : ''} ${r.score} ${r.timeSeconds} giây ${r.correct} / ${r.total} ${subject} (${r.mode || mode}) `; tbody.appendChild(tr); rank++; }); document.getElementById('leaderboard-table').classList.remove('hidden'); document.getElementById('leaderboard-info').textContent = `Top ${Math.min(50, results.length)} kết quả môn ${subject}`; document.getElementById('leaderboard-info').classList.remove('hidden'); document.getElementById('leaderboard-empty').classList.add('hidden'); } catch (err) { console.error("Lỗi tải BXH:", err); tbody.innerHTML = 'Lỗi tải dữ liệu (xem console)'; } }
